project(HARE)

set(HARE_BUILD_DIR ${HARE_BINARY_DIR}/build)

set(HARE_BUILD_ADDONS_DIR ${HARE_BINARY_DIR}/build/add-ons)

include(UseHaiku)

add_subdirectory(src)
